const darkenColor = (hex, percent) => {
    const num = parseInt(hex.replace(/^#/, ''), 16);
    const amt = Math.round(2.55 * percent);
    return (((num >> 16) - amt) < 0 ? 0 : ((num >> 16) - amt))
        .toString(16)
        .padStart(2, 0) +
        ((((num >> 8) & 0x00ff) - amt) < 0 ? 0 : (((num >> 8) & 0x00ff) - amt))
            .toString(16)
            .padStart(2, 0) +
            (((num & 0x0000ff) - amt) < 0 ? 0 : ((num & 0x0000ff) - amt))
                .toString(16)
                .padStart(2, 0) || hex;
};

const Folder = ({
    color = '#5227FF',
    size = 1,
    items = [],
    className = ''
}) => {
    const [open, setOpen] = useState(false);
    const [paperOffsets, setPaperOffsets] = useState([
        { x: 0, y: 0 },
        { x: 0, y: 0 },
        { x: 0, y: 0 }
    ]);

    const handleClick = () => {
        setOpen(!open);
    };

    const handlePaperMouseMove = (e) => {
        const papers = document.querySelectorAll('.paper');
        papers.forEach((paper, index) => {
            const rect = paper.getBoundingClientRect();
            const centerX = rect.left + rect.width / 2;
            const centerY = rect.top + rect.height / 2;
            const distX = e.clientX - centerX;
            const distY = e.clientY - centerY;
            const distance = Math.sqrt(distX * distX + distY * distY);
            const maxDistance = 200;

            if (distance < maxDistance) {
                const angle = Math.atan2(distY, distX);
                const force = (maxDistance - distance) / maxDistance;
                const offsetX = Math.cos(angle) * force * 30;
                const offsetY = Math.sin(angle) * force * 30;

                setPaperOffsets(prev => {
                    const newOffsets = [...prev];
                    newOffsets[index] = { x: offsetX, y: offsetY };
                    return newOffsets;
                });
            }
        });
    };

    const handlePaperMouseLeave = () => {
        setPaperOffsets([
            { x: 0, y: 0 },
            { x: 0, y: 0 },
            { x: 0, y: 0 }
        ]);
    };

    return (
        <div
            className={`folder ${open ? 'open' : ''} ${className}`}
            onClick={handleClick}
            onMouseMove={handlePaperMouseMove}
            onMouseLeave={handlePaperMouseLeave}
            style={{
                '--folder-color': color,
                '--folder-back-color': darkenColor(color.slice(1), 20),
                '--paper-1': '#FFE5E5',
                '--paper-2': '#E5F0FF',
                '--paper-3': '#FFFAE5',
                '--magnet-x': `${paperOffsets[0]?.x || 0}px`,
                '--magnet-y': `${paperOffsets[0]?.y || 0}px`,
                transform: `scale(${size})`
            }}
        >
            <div className="folder__back">
                {[0, 1, 2].map((i) => (
                    <div
                        key={i}
                        className={`paper paper-${i + 1}`}
                        style={{
                            '--magnet-x': `${paperOffsets[i]?.x || 0}px`,
                            '--magnet-y': `${paperOffsets[i]?.y || 0}px`,
                        }}
                    />
                ))}
                <div className="folder__front" />
                <div className="folder__front right" />
            </div>
        </div>
    );
};

export default Folder;
